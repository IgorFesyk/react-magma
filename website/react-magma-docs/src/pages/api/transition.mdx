<DocsHeading type="api">Transition</DocsHeading>

Smart wrappers using framer-motion to add transitions to child components.

## Basic Usage

      'fade',
      'slideTop',
      'slideBottom',
      'slideRight',
      'slideLeft',
      'scale',
      'collapse'

### Fade

```typescript
import React from 'react';
import { Transition } from 'react-magma-dom';

export function Example() {
  const [isOpen, setIsOpen] = React.useState(false);
  return (
    <div>
      <Transition in={isOpen} fade style={{ overflow: 'hidden' }}>
        <div
          style={{
            height: '250px',
            width: '250px',
            backgroundColor: '#bada55',
          }}
        />
      </Transition>
      <Button onClick={() => setIsOpen(!isOpen)}>
        {isOpen ? 'hide' : 'show'}
      </Button>
    </div>
  );
}
```

### Composition

Transitions can be combined to create more advanced effects.

```typescript
import React from 'react';
import { Transition } from 'react-magma-dom';

export function Example() {
  const [isOpen, setIsOpen] = React.useState(false);
  return (
    <div>
      <Transition in={isOpen} scale fade style={{ overflow: 'hidden' }}>
        <div
          style={{
            height: '250px',
            width: '250px',
            backgroundColor: '#bada55',
          }}
        />
      </Transition>
      <Button onClick={() => setIsOpen(!isOpen)}>
        {isOpen ? 'hide' : 'show'}
      </Button>
    </div>
  );
}
```
