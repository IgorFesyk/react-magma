## Controlled

<Playground>
  {() => {
    class SelectInput extends React.Component {
      constructor(props) {
        super(props)
          this.state = {
            value: null
          };

          this.changeValue = this.changeValue.bind(this);
        }

        changeValue(value) {
          this.setState(state => ({
            value
          }));
        }

        render() {
          return <div>
            <p>
              <strong>Value: </strong>
              <span id="value">{this.state.value ? this.state.value.value : 'No Value Set'}</span>
            </p>
            <Select
              id="valueSelectId"
              name="value"
              labelText="Value"
              options={[{
                value: 'red',
                label: 'Red'
              }, {
                value: 'blue',
                label: 'Blue'
              }, {
                value: 'green',
                label: 'Green'
              }]}
              handleChange={this.changeValue}
            />
          </div>
        }
      }
      return <SelectInput />
  }}
</Playground>

## Focus Events

<Playground>
  {() => {
    class SelectInput extends React.Component {
      constructor(props) {
        super(props)
          this.state = {
            focused: false,
            message: 'No'
          };

          this.changeFocusedMessage = this.changeFocusedMessage.bind(this);
        }

        changeFocusedMessage() {
          const focused = !this.state.focused;

          this.setState(state => ({
            focused,
            message: focused ? 'Yes' : 'No',
          }));
        }

        render() {
          return <div>
            <p><strong>Focused: </strong> <span id="focused">{this.state.message}</span></p>
            <Select
              id="focusedSelectId"
              name="focused"
              labelText="Focused?"
              options={[{
                value: 'red',
                label: 'Red'
              }, {
                value: 'blue',
                label: 'Blue'
              }, {
                value: 'green',
                label: 'Green'
              }]}
              handleFocus={this.changeFocusedMessage}
              handleBlur={this.changeFocusedMessage}
            />
          </div>
        }
      }
      return <SelectInput />
  }}
</Playground>
