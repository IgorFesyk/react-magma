### Click Events

<Playground>
  {() => {
    class Example extends React.Component {
      constructor(props) {
        super(props)
          this.state = {
            counter: 0
          }
        }

        render() {
          return <div>
          <p><strong>Counter: </strong> <span id="counter">{this.state.counter}</span></p>
          <Icon
            handleClick={() => {
              this.setState(state => ({counter: state.counter + 1}))
            }}
            id="clickIcon"
            title="On Click Icon"
            type="angle-down"
          />
          </div>
        }
      }
      return <Example />
  }}
</Playground>
