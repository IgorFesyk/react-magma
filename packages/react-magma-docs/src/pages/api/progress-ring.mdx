# Progress Ring

## Basic Usage

```.jsx
(props) => {
    const [percentage, setPercentage] = React.useState(0);

    React.useEffect(() => {
       const timer1 = setTimeout(() => {
        setPercentage(20);
      }, 2000);

      const timer2 = setTimeout(() => {
        setPercentage(40);
      }, 4000);

      const timer3 = setTimeout(() => {
        setPercentage(60);
      }, 6000);

      const timer4 = setTimeout(() => {
        setPercentage(80);
      }, 8000);

      const timer5 = setTimeout(() => {
        setPercentage(100);
      }, 10000);

      return () => {
        clearTimeout(timer1);
        clearTimeout(timer2);
        clearTimeout(timer3);
        clearTimeout(timer4);
        clearTimeout(timer5);
      }
    }, [])

    return (
        <ProgressRing percentage={percentage} />
      )
  }
```
