## Table Props

**Any other props supplied will be provided to the wrapping `table` element.**

<SimplePropsTable
  propertyValues={{
    children: {
      type: {
        name: 'node',
      },
      required: true,
      description: 'The content of the component',
    },
    density: {
      type: {
        name: 'enum',
        options: [
          'TableDensity.compact',
          'TableDensity.loose',
          'TableDensity.normal',
        ],
      },
      required: false,
      description: 'Relative padding of the table cells',
      defaultValue: 'normal',
    },
    hasVerticalBorders: {
      type: {
        name: 'boolean',
      },
      required: false,
      description: 'If true, columns will have vertical borders',
    },
    hasZebraStripes: {
      type: {
        name: 'boolean',
      },
      required: false,
      description: 'If true, every other row will have a background color',
    },
    isInverse: {
      type: {
        name: 'boolean',
      },
      required: false,
      description:
        'If true, the component will have inverse styling to better appear on a dark background',
      defaultValue: 'false',
    },
  }}
/>

## TableCell / TableHeaderCell Props

**Any other props supplied will be provided to the wrapping `th` or `td` element.**

  propertyValues={{
